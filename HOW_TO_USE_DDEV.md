```bash
# 새 프로젝트 구성
ddev config

# 프로젝트 시작
ddev start

# 프로젝트 중지
ddev stop

# 모든 프로젝트 중지
ddev poweroff

# 프로젝트 상태 확인
ddev status

# 프로젝트 정보 확인
ddev describe

# 프로젝트 컨테이너에 SSH 접속
ddev ssh

# 데이터베이스 관리 도구 열기
ddev launch -p

# 프로젝트 삭제 (컨테이너와 볼륨 제거)
ddev delete

# 프로젝트 중지 및 목록에서 제거 (데이터 유지)
ddev stop --unlist
```
